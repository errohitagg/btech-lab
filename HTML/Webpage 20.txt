<html>
<head>
<title>Webpage 20</title>
<script language="Javascript">
var i;
i=prompt("Enter the number for the day of the week : ");
switch(i)
 {
  case '1':document.write("<h2>The day is Sunday</h2>");
         break;
  case '2':document.write("<h2>The day is Monday</h2>");
         break;
  case '3':document.write("<h2>The day is Tuesday</h2>");
         break;
  case '4':document.write("<h2>The day is Wednesday</h2>");
         break;
  case '5':document.write("<h2>The day is Thursday</h2>");
         break;
  case '6':document.write("<h2>The day is Friday</h2>");
         break;
  case '7':document.write("<h2>The day is Saturday</h2>");
         break;
  default:document.write("<h3>Wrong number entered.<br>"+
                         "No corresponding day for the number.</h3>");
          break;
 }
</script>
</head>
